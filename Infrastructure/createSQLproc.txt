CREATE OR REPLACE PROCEDURE runSQL(in SQLsource NVARCHAR(5000), SQLStmt NCLOB) AS BEGIN    
    DECLARE MYCOND CONDITION FOR SQL_ERROR_CODE 10001;
	DECLARE EXIT HANDLER FOR SQLEXCEPTION SELECT :SQLNumber as "SQL source", ::SQL_ERROR_CODE as "Return code", ::SQL_ERROR_MESSAGE as "Message" FROM DUMMY;

	EXEC :SQLstmt;
	
	SIGNAL MYCOND SET MESSAGE_TEXT = 'OK';
END;
